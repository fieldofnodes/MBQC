41std::vector<int> getVertexNeighbours(const undirectedGraph& graph,int vertex){
42 // Choose a vertex to find its adjacent vertices
43 undirectedGraph::vertex_descriptor v = vertex;
44 // Find the range of adjacent vertices
45 typedef boost::graph_traits<undirectedGraph>::adjacency_iterator adjacency_iterator;
46 std::pair<adjacency_iterator, adjacency_iterator> neighbors = adjacent_vertices(v, graph);
47
48 std::vector<int> neighbourVector;
49
50 for (adjacency_iterator it = neighbors.first; it != neighbors.second; ++it) {
51 neighbourVector.push_back(*it);
52 }
53 return neighbourVector;
54}

76qreal computeCorrectedAngle(
77 const undirectedGraph& graph,
78 std::vector<int> measuredOutcomes,
79 int vertex,
80 int numRows,
81 qreal angle){
82
83 std::vector<int> neighbourhood = getVertexNeighbours(graph,vertex);
84 int xCorrectionVertex;
85 std::vector<int> zCorrectionVertices;
86 qreal updatedAngle;
87
88 if (vertex < numRows){ // First column - no x correction no z correction
89 updatedAngle = angle;
90 } else if (vertex >= numRows){ // ensure we are in second column
91 xCorrectionVertex = getLattice2DPastVertexGraph(graph,vertex,numRows);
92 for (int i: neighbourhood){
93 int past;
94 if(i >= numRows){
95 past = getLattice2DPastVertexGraph(graph,i,numRows);
96 } else{
97 continue;
98 }
99 zCorrectionVertices.push_back(past);
100 }
101
102 // Remove vertex in neigbourhood if it is the vertrex
103 // we want to measure
104 zCorrectionVertices.erase(
105 std::remove(
106 zCorrectionVertices.begin(),
107 zCorrectionVertices.end(), vertex),
108 zCorrectionVertices.end());
109
110 int exponent = measuredOutcomes[xCorrectionVertex];
111
112 //std::cout << "f-1(j): " << xCorrectionVertex << std::endl;
113 //std::cout << "s_f-1(j): " << exponent << std::endl;
114 qreal xAngle = std::pow((-1), exponent)*angle;
115
116 std::vector<int> outcomesForZ;
117 for (int i:zCorrectionVertices){
118 outcomesForZ.push_back(measuredOutcomes[i]);
119 }
120
121 int sum = std::accumulate(outcomesForZ.begin(), outcomesForZ.end(), 0);
122 qreal zAngle = M_PI*sum;
123 updatedAngle = xAngle + zAngle;
124
125 }
126
127 return updatedAngle;
